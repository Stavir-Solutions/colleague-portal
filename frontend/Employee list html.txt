    <script>
        const editButtons = document.querySelectorAll('.edit-button');
        const saveButtons = document.querySelectorAll('.save-button');
        const editableCells = document.querySelectorAll('td[contenteditable="true"]');
        const addRowButton = document.getElementById('add-row-button');
        const deleteRowButton = document.getElementById('delete-row-button');

        editButtons.forEach((editButton, index) => {
            editButton.addEventListener('click', () => {
                editableCells.forEach(cell => {
                    cell.setAttribute('contenteditable', 'true');
                });

                editButtons[index].style.display = 'none';
                saveButtons[index].style.display = 'block';
            });
        });

        saveButtons.forEach((saveButton, index) => {
            saveButton.addEventListener('click', () => {
                editableCells.forEach(cell => {
                    cell.setAttribute('contenteditable', 'false');
                });

                editButtons[index].style.display = 'block';
                saveButtons[index].style.display = 'none';
            });
        });

        addRowButton.addEventListener('click', () => {
            const newRow = document.createElement('tr');
            newRow.innerHTML = `
                <td contenteditable="true"></td>
                <td contenteditable="true"></td>
                <td contenteditable="true"></td>
                <td contenteditable="true"></td>
                <td contenteditable="true"></td>
                <td>
                    <button class="edit-button">Edit</button>
                    <button class="save-button" style="display: none;">Save</button>
                </td>
            `;

            const newEditButton = newRow.querySelector('.edit-button');
            const newSaveButton = newRow.querySelector('.save-button');
            const newEditableCells = newRow.querySelectorAll('td[contenteditable="true"]');

            newEditButton.addEventListener('click', () => {
                newEditableCells.forEach(cell => {
                    cell.setAttribute('contenteditable', 'true');
                });

                newEditButton.style.display = 'none';
                newSaveButton.style.display = 'block';
            });

            newSaveButton.addEventListener('click', () => {
                newEditableCells.forEach(cell => {
                    cell.setAttribute('contenteditable', 'false');
                });

                newEditButton.style.display = 'block';
                newSaveButton.style.display = 'none';
            });

            document.querySelector('tbody').appendChild(newRow);
        });

        deleteRowButton.addEventListener('click', () => {
            const rowCount = document.querySelectorAll('tbody tr').length;
            if (rowCount > 1) {
                document.querySelector('tbody tr:last-child').remove();
            }
        });
    </script>
